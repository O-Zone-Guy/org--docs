  import re
  from fnmatch import fnmatch
  from os import listdir

  FILES = [f for f in listdir('.') if fnmatch(f, '[!\.\*]*.org')]

  r = re.compile('#\+TITLE: (.*)', re.I)

  lists = ""
  for f in FILES:
      fr = open(f, 'r')
      try:
          lists += " - [[file:" + f + "][" + r.match(fr.read()).group(1) + "]]\n"
          # lists +=  r.match(fr.read()).group(1) + "\n"
      except:
          pass

  return lists
